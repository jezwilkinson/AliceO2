42  Configurable<std::string> paramfile{"param-file", "", "Path to the parametrization object, if emtpy the parametrization is not taken from file"};
43  Configurable<std::string> signalname{"param-signal", "BetheBloch", "Name of the parametrization for the expected signal, used in both file and CCDB mode"};
44  Configurable<std::string> sigmaname{"param-sigma", "TPCReso", "Name of the parametrization for the expected sigma, used in both file and CCDB mode"};
45  Configurable<std::string> url{"ccdb-url", "http://ccdb-test.cern.ch:8080", "url of the ccdb repository"};

48  void init(o2::framework::InitContext&)
49  {
50  ccdb->setURL(url.value);
51  ccdb->setTimestamp(timestamp.value);
52  ccdb->setCaching(true);
53  ccdb->setLocalObjectValidityChecking();
54  // Not later than now objects
55  ccdb->setCreatedNotAfter(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
56  //
57  const std::string fname = paramfile.value;
58  if (!fname.empty()) { // Loading the parametrization from file
59  resp.LoadParamFromFile(fname.data(), signalname.value, DetectorResponse::kSignal);
60  resp.LoadParamFromFile(fname.data(), sigmaname.value, DetectorResponse::kSigma);
61  } else { // Loading it from CCDB
62  const std::string path = "Analysis/PID/TPC";
63  resp.LoadParam(DetectorResponse::kSignal, ccdb->getForTimeStamp<Parametrization>(path + "/" + signalname.value, timestamp.value));
64  resp.LoadParam(DetectorResponse::kSigma, ccdb->getForTimeStamp<Parametrization>(path + "/" + sigmaname.value, timestamp.value));
65  }
66  }

242  expected_diff->Fill<Al>(mom, i.tpcSignal() - i.tpcExpSignalAl());
