114  // Version with explicit nested loop
115  void process(soa::Filtered<soa::Join<aod::Collisions, aod::EvSels, aod::Cents>>::iterator const& collision, myTracks const& tracks)
116  {
117  LOGF(info, "Tracks for collision: %d | Vertex: %.1f | INT7: %d | V0M: %.1f", tracks.size(), collision.posZ(), collision.sel7(), collision.centV0M());
118 
119  const auto centrality = collision.centV0M();
120 
121  same->fillEvent(centrality, CorrelationContainer::kCFStepAll);
122 
123  if (!collision.sel7())
124  return;
125 
126  same->fillEvent(centrality, CorrelationContainer::kCFStepTriggered);
127 
128  // vertex already checked as filter
129  same->fillEvent(centrality, CorrelationContainer::kCFStepVertex);
130 
131  same->fillEvent(centrality, CorrelationContainer::kCFStepReconstructed);
132 
133  int bSign = 1; // TODO magnetic field from CCDB
134 
135  for (auto& track1 : tracks) {
136 
137  // LOGF(info, "Track %f | %f | %f %d %d", track1.eta(), track1.phi(), track1.pt(), track1.isGlobalTrack(), track1.isGlobalTrackSDD());
138 
139  // control histograms
140  // ((TH3*) (registry.get("yields").get()))->Fill(centrality, track1.pt(), track1.eta());
141  // ((TH3*) (registry.get("etaphi").get()))->Fill(centrality, track1.eta(), track1.phi());
142  yields->Fill(centrality, track1.pt(), track1.eta());
143  etaphi->Fill(centrality, track1.eta(), track1.phi());
144 
145  if (cfgTriggerCharge != 0 && cfgTriggerCharge * track1.charge() < 0)
146  continue;
147 
148  double eventValues[3];
149  eventValues[0] = track1.pt();
150  eventValues[1] = centrality;
151  eventValues[2] = collision.posZ();
152 
153  same->getTriggerHist()->Fill(eventValues, CorrelationContainer::kCFStepReconstructed);
154  //mixed->getTriggerHist()->Fill(eventValues, CorrelationContainer::kCFStepReconstructed);
155 
156  for (auto& track2 : tracks) {
157  if (track1 == track2)
158  continue;
159 
160  if (cfgPtOrder != 0 && track2.pt() >= track1.pt())
161  continue;
162 
163  if (cfgAssociatedCharge != 0 && cfgAssociatedCharge * track2.charge() < 0)
164  continue;
165  if (cfgPairCharge != 0 && cfgPairCharge * track1.charge() * track2.charge() < 0)
166  continue;
167 
168  if (cfg.mPairCuts && mPairCuts.conversionCuts(track1, track2))
169  continue;
170 
171  if (cfgTwoTrackCut > 0 && mPairCuts.twoTrackCut(track1, track2, bSign))
172  continue;
173 
174  double values[6] = {0};
175 
176  values[0] = track1.eta() - track2.eta();
177  values[1] = track2.pt();
178  values[2] = track1.pt();
179  values[3] = centrality;
180 
181  values[4] = track1.phi() - track2.phi();
182  if (values[4] > 1.5 * TMath::Pi())
183  values[4] -= TMath::TwoPi();
184  if (values[4] < -0.5 * TMath::Pi())
185  values[4] += TMath::TwoPi();
186 
187  values[5] = collision.posZ();
188 
189  same->getPairHist()->Fill(values, CorrelationContainer::kCFStepReconstructed);
190  //mixed->getPairHist()->Fill(values, CorrelationContainer::kCFStepReconstructed);
191  }
192  }
193  }

197  {
198  LOGF(info, "Tracks for collision (Combination run): %d", tracks.size());
199 
200  int bSign = 1; // TODO magnetic field from CCDB
201 
202  for (auto track1 = tracks.begin(); track1 != tracks.end(); ++track1) {
203 
204  if (cfgTriggerCharge != 0 && cfgTriggerCharge * track1.charge() < 0)
205  continue;
206 
207  // LOGF(info, "TRACK %f %f | %f %f | %f %f", track1.eta(), track1.eta(), track1.phi(), track1.phi2(), track1.pt(), track1.pt());
208 
209  double eventValues[3];
210  eventValues[0] = track1.pt();
211  eventValues[1] = 0; // collision.v0mult();
212  eventValues[2] = collision.posZ();
213 
214  same->getTriggerHist()->Fill(eventValues, CorrelationContainer::kCFStepReconstructed);
215  //mixed->getTriggerHist()->Fill(eventValues, CorrelationContainer::kCFStepReconstructed);
216  }
217 
218  for (auto& [track1, track2] : combinations(tracks, tracks)) {
219  //LOGF(info, "Combination %d %d", track1.index(), track2.index());
220 
221  if (cfgTriggerCharge != 0 && cfgTriggerCharge * track1.charge() < 0)
222  continue;
223  if (cfgAssociatedCharge != 0 && cfgAssociatedCharge * track2.charge() < 0)
224  continue;
225  if (cfgPairCharge != 0 && cfgPairCharge * track1.charge() * track2.charge() < 0)
226  continue;
227 
228  if (cfg.mPairCuts && mPairCuts.conversionCuts(track1, track2))
229  continue;
230 
231  if (cfgTwoTrackCut > 0 && mPairCuts.twoTrackCut(track1, track2, bSign))
232  continue;
233 
234  double values[6] = {0};
235 
236  values[0] = track1.eta() - track2.eta();
237  values[1] = track1.pt();
238  values[2] = track2.pt();
239  values[3] = 0; // collision.v0mult();
