195  // create histogram of type T with the axes defined in HistogramSpec
196  template <typename T>
197  static std::shared_ptr<T> createHist(const HistogramSpec& histSpec)
198  {
199  constexpr std::size_t MAX_DIM{10};
200  const std::size_t nAxes{histSpec.config.axes.size()};
201  if (nAxes == 0 || nAxes > MAX_DIM) {
202  LOGF(FATAL, "The histogram specification contains no (or too many) axes.");
203  return nullptr;
204  }
205 
206  int nBins[MAX_DIM]{0};
207  double lowerBounds[MAX_DIM]{0.};
208  double upperBounds[MAX_DIM]{0.};
209 
210  // first figure out number of bins and dimensions
211  for (std::size_t i = 0; i < nAxes; i++) {
212  nBins[i] = (histSpec.config.axes[i].nBins) ? *histSpec.config.axes[i].nBins : histSpec.config.axes[i].binEdges.size() - 1;
213  lowerBounds[i] = histSpec.config.axes[i].binEdges.front();
214  upperBounds[i] = histSpec.config.axes[i].binEdges.back();
215  }
216 
217  // create histogram
218  std::shared_ptr<T> hist{generateHist<T>(histSpec.name, histSpec.title, nAxes, nBins, lowerBounds, upperBounds)};
219  if (!hist) {
220  LOGF(FATAL, "The number of specified dimensions does not match the type.");
221  return nullptr;
222  }
223 
224  // set axis properties
225  for (std::size_t i = 0; i < nAxes; i++) {
226  TAxis* axis{getAxis(i, hist)};
227  if (axis) {
228  if (histSpec.config.axes[i].title)
229  axis->SetTitle((*histSpec.config.axes[i].title).data());
230 
231  // this helps to have axes not only called 0,1,2... in ndim histos
232  if constexpr (std::is_base_of_v<THnBase, T>) {
233  if (histSpec.config.axes[i].name)
234  axis->SetName((std::string(axis->GetName()) + "-" + *histSpec.config.axes[i].name).data());
235  }
236 
237  // move the bin edges in case a variable binning was requested
238  if (!histSpec.config.axes[i].nBins) {
239  if (!std::is_sorted(std::begin(histSpec.config.axes[i].binEdges), std::end(histSpec.config.axes[i].binEdges))) {
240  LOGF(FATAL, "The bin edges in histogram %s are not in increasing order!", histSpec.name);
241  return nullptr;
242  }
243  axis->Set(nBins[i], histSpec.config.axes[i].binEdges.data());
244  }
245  }
246  }
247  if (histSpec.callSumw2)
248  hist->Sumw2();
249 
250  return hist;
251  }

275  // helper function to generate the actual histograms
276  template <typename T>
277  static T* generateHist(const std::string& name, const std::string& title, const std::size_t nDim,
278  const int nBins[], const double lowerBounds[], const double upperBounds[])
279  {
280  if constexpr (std::is_base_of_v<THnBase, T>) {
281  return new T(name.data(), title.data(), nDim, nBins, lowerBounds, upperBounds);
282  } else if constexpr (std::is_base_of_v<TH3, T>) {
283  return (nDim != 3) ? nullptr
284  : new T(name.data(), title.data(), nBins[0], lowerBounds[0],
285  upperBounds[0], nBins[1], lowerBounds[1], upperBounds[1],
286  nBins[2], lowerBounds[2], upperBounds[2]);
287  } else if constexpr (std::is_base_of_v<TH2, T>) {
288  return (nDim != 2) ? nullptr
289  : new T(name.data(), title.data(), nBins[0], lowerBounds[0],
290  upperBounds[0], nBins[1], lowerBounds[1], upperBounds[1]);
291  } else if constexpr (std::is_base_of_v<TH1, T>) {
292  return (nDim != 1)
293  ? nullptr
294  : new T(name.data(), title.data(), nBins[0], lowerBounds[0], upperBounds[0]);
295  }
296  return nullptr;
297  }
298 
299  // helper function to get the axis via index for any type of root histogram
300  template <typename T>
301  static TAxis* getAxis(const int i, std::shared_ptr<T>& hist)
302  {
303  if constexpr (std::is_base_of_v<THnBase, T>) {
304  return hist->GetAxis(i);
305  } else {
306  return (i == 0) ? hist->GetXaxis()
307  : (i == 1) ? hist->GetYaxis() : (i == 2) ? hist->GetZaxis() : nullptr;
308  }
309  }
310 
311  // helper function to cast the actual histogram type (e.g. TH2F) to the correct interface type (e.g. TH2) that is stored in the HistPtr variant
312  template <typename T>
313  static std::optional<HistPtr> castToVariant(std::shared_ptr<TObject> obj)
314  {
315  if (obj->InheritsFrom(T::Class())) {
316  return std::static_pointer_cast<T>(obj);
317  }
318  return std::nullopt;
319  }
320  template <typename T, typename Next, typename... Rest>
321  static std::optional<HistPtr> castToVariant(std::shared_ptr<TObject> obj)
322  {
323  if (auto hist = castToVariant<T>(obj)) {
324  return hist;
325  }
326  return castToVariant<Next, Rest...>(obj);
327  }
328  static std::optional<HistPtr> castToVariant(std::shared_ptr<TObject> obj)
329  {
330  if (obj) {
331  // TProfile3D is TH3, TProfile2D is TH2, TH3 is TH1, TH2 is TH1, TProfile is TH1
332  return castToVariant<THn, THnSparse, TProfile3D, TH3, TProfile2D, TH2, TProfile, TH1>(obj);
333  }
334  return std::nullopt;
335  }

344 struct HistFiller {
345  // fill any type of histogram (if weight was requested it must be the last argument)
346  template <bool useWeight = false, typename T, typename... Ts>
347  static void fillHistAny(std::shared_ptr<T>& hist, const Ts&... positionAndWeight)
348  {
349  constexpr int nDim = sizeof...(Ts) - useWeight;
350 
351  constexpr bool validTH3 = (std::is_same_v<TH3, T> && nDim == 3);
352  constexpr bool validTH2 = (std::is_same_v<TH2, T> && nDim == 2);
353  constexpr bool validTH1 = (std::is_same_v<TH1, T> && nDim == 1);
354  constexpr bool validTProfile3D = (std::is_same_v<TProfile3D, T> && nDim == 4);
355  constexpr bool validTProfile2D = (std::is_same_v<TProfile2D, T> && nDim == 3);
356  constexpr bool validTProfile = (std::is_same_v<TProfile, T> && nDim == 2);
357 
358  constexpr bool validSimpleFill = validTH1 || validTH2 || validTH3 || validTProfile || validTProfile2D || validTProfile3D;
359  // unfortunately we dont know at compile the dimension of THn(Sparse)
360  constexpr bool validComplexFill = std::is_base_of_v<THnBase, T>;
361 
362  if constexpr (validSimpleFill) {
363  hist->Fill(static_cast<double>(positionAndWeight)...);
364  } else if constexpr (validComplexFill) {
365  // savety check for n dimensional histograms (runtime overhead)
366  if (hist->GetNdimensions() != nDim)
367  throw runtime_error("The number of position (and weight) arguments in fill() does not match the histogram dimensions!");
368 
369  double tempArray[sizeof...(Ts)] = {static_cast<double>(positionAndWeight)...};
370  if constexpr (useWeight)
371  hist->Fill(tempArray, tempArray[sizeof...(Ts) - 1]);
372  else
373  hist->Fill(tempArray);
374  } else {
375  throw runtime_error("The number of position (and weight) arguments in fill() does not match the histogram dimensions!");
376  }
377  }
378 
379  // fill any type of histogram with columns (Cs) of a filtered table (if weight is requested it must reside the last specified column)
380  template <bool useWeight = false, typename... Cs, typename R, typename T>
381  static void fillHistAnyTable(std::shared_ptr<R>& hist, const T& table, const o2::framework::expressions::Filter& filter)
382  {
383  auto filtered = o2::soa::Filtered<T>{{table.asArrowTable()}, o2::framework::expressions::createSelection(table.asArrowTable(), filter)};
384  for (auto& t : filtered) {
385  fillHistAny<useWeight>(hist, (*(static_cast<Cs>(t).getIterator()))...);
386  }
387  }
388 };

425  // gets the underlying histogram pointer
426  // we cannot automatically infer type here so it has to be explicitly specified
427  // -> get<TH1>(), get<TH2>(), get<TH3>(), get<THn>(), get<THnSparse>(), get<TProfile>(), get<TProfile2D>(), get<TProfile3D>()
429  template <typename T>
430  auto& get(char const* const name)
431  {
432  const uint32_t id = compile_time_hash(name);
433  const uint32_t i = imask(id);
434  if (O2_BUILTIN_LIKELY(id == mRegistryKey[i])) {
435  if (auto histPtr = std::get_if<std::shared_ptr<T>>(&mRegistryValue[i]))
436  return *histPtr;
437  else
438  throw runtime_error("Histogram type specified in get() does not match actual histogram type!");
439  }
440  for (auto j = 1u; j < MAX_REGISTRY_SIZE; ++j) {
441  if (id == mRegistryKey[imask(j + i)]) {
442  if (auto histPtr = std::get_if<std::shared_ptr<T>>(&mRegistryValue[imask(j + i)]))
443  return *histPtr;
444  else
445  throw runtime_error("Histogram type specified in get() does not match actual histogram type!");
446  }
447  }
448  throw runtime_error("No matching histogram found in HistogramRegistry!");
449  }

490  // fill hist with values
491  template <bool useWeight = false, typename... Ts>
492  void fill(char const* const name, Ts&&... positionAndWeight)
493  {
494  const uint32_t id = compile_time_hash(name);
495  const uint32_t i = imask(id);
496  if (O2_BUILTIN_LIKELY(id == mRegistryKey[i])) {
497  std::visit([this, &positionAndWeight...](auto&& hist) { HistFiller::fillHistAny<useWeight>(hist, std::forward<Ts>(positionAndWeight)...); }, mRegistryValue[i]);
498  return;
499  }
500  for (auto j = 1u; j < MAX_REGISTRY_SIZE; ++j) {

502  std::visit([this, &positionAndWeight...](auto&& hist) { HistFiller::fillHistAny<useWeight>(hist, std::forward<Ts>(positionAndWeight)...); }, mRegistryValue[imask(j + i)]);
503  return;
504  }

527  // this is for internal use only because we dont want user to have to specify 'useWeight' argument
528  template <bool useWeight = false, typename... Cs, typename T>
529  void fillTable(char const* const name, const T& table, const o2::framework::expressions::Filter& filter)
530  {

532  const uint32_t i = imask(id);

534  std::visit([this, &table, &filter](auto&& hist) { HistFiller::fillHistAnyTable<useWeight, Cs...>(hist, table, filter); }, mRegistryValue[i]);
535  return;

538  if (id == mRegistryKey[imask(j + i)]) {
539  std::visit([this, &table, &filter](auto&& hist) { HistFiller::fillHistAnyTable<useWeight, Cs...>(hist, table, filter); }, mRegistryValue[imask(j + i)]);
540  return;
541  }

550  void insert(const HistogramSpec& histSpec)
551  {
552  uint32_t i = imask(histSpec.id);
553  for (auto j = 0u; j < MAX_REGISTRY_SIZE; ++j) {
554  TObject* rawPtr = nullptr;
555  std::visit([&](const auto& sharedPtr) { rawPtr = sharedPtr.get(); }, mRegistryValue[imask(j + i)]);
556  if (!rawPtr) {
557  mRegistryKey[imask(j + i)] = histSpec.id;
558  mRegistryValue[imask(j + i)] = HistFactory::createHistVariant(histSpec);
559  lookup += j;
560  return;
561  }
562  }
563  throw runtime_error("Internal array of HistogramRegistry is full.");
564  }
