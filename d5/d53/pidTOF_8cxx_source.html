45 struct pidTOFTask {
46  using Trks = soa::Join<aod::Tracks, aod::TracksExtra>;
47  using Coll = aod::Collision;
48  Produces<aod::pidRespTOF> tofpid;
49  DetectorResponse resp;
50  Service<o2::ccdb::BasicCCDBManager> ccdb;
51  Configurable<std::string> paramfile{"param-file", "", "Path to the parametrization object, if emtpy the parametrization is not taken from file"};
52  Configurable<std::string> sigmaname{"param-sigma", "TOFReso", "Name of the parametrization for the expected sigma, used in both file and CCDB mode"};
53  Configurable<std::string> url{"ccdb-url", "http://ccdb-test.cern.ch:8080", "url of the ccdb repository"};
54  Configurable<long> timestamp{"ccdb-timestamp", -1, "timestamp of the object"};
55 
56  void init(o2::framework::InitContext&)
57  {
58  ccdb->setURL(url.value);
59  ccdb->setTimestamp(timestamp.value);
60  ccdb->setCaching(true);
61  ccdb->setLocalObjectValidityChecking();
62  // Not later than now objects
63  ccdb->setCreatedNotAfter(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
64  //
65  const std::vector<float> p = {0.008, 0.008, 0.002, 40.0};
66  resp.SetParameters(DetectorResponse::kSigma, p);
67  const std::string fname = paramfile.value;
68  if (!fname.empty()) { // Loading the parametrization from file
69  resp.LoadParamFromFile(fname.data(), sigmaname.value, DetectorResponse::kSigma);
70  } else { // Loading it from CCDB
71  const std::string path = "Analysis/PID/TOF";
72  resp.LoadParam(DetectorResponse::kSigma, ccdb->getForTimeStamp<Parametrization>(path + "/" + sigmaname.value, timestamp.value));
73  }
74  }

76  void process(Coll const& collision, Trks const& tracks)
77  {
78  constexpr tof::ExpTimes<Coll, Trks::iterator, PID::Electron> resp_Electron = tof::ExpTimes<Coll, Trks::iterator, PID::Electron>();
79  constexpr tof::ExpTimes<Coll, Trks::iterator, PID::Muon> resp_Muon = tof::ExpTimes<Coll, Trks::iterator, PID::Muon>();
80  constexpr tof::ExpTimes<Coll, Trks::iterator, PID::Pion> resp_Pion = tof::ExpTimes<Coll, Trks::iterator, PID::Pion>();
81  constexpr tof::ExpTimes<Coll, Trks::iterator, PID::Kaon> resp_Kaon = tof::ExpTimes<Coll, Trks::iterator, PID::Kaon>();
82  constexpr tof::ExpTimes<Coll, Trks::iterator, PID::Proton> resp_Proton = tof::ExpTimes<Coll, Trks::iterator, PID::Proton>();
83  constexpr tof::ExpTimes<Coll, Trks::iterator, PID::Deuteron> resp_Deuteron = tof::ExpTimes<Coll, Trks::iterator, PID::Deuteron>();
84  constexpr tof::ExpTimes<Coll, Trks::iterator, PID::Triton> resp_Triton = tof::ExpTimes<Coll, Trks::iterator, PID::Triton>();
85  constexpr tof::ExpTimes<Coll, Trks::iterator, PID::Helium3> resp_Helium3 = tof::ExpTimes<Coll, Trks::iterator, PID::Helium3>();
86  constexpr tof::ExpTimes<Coll, Trks::iterator, PID::Alpha> resp_Alpha = tof::ExpTimes<Coll, Trks::iterator, PID::Alpha>();
87 
88  tofpid.reserve(tracks.size());
89  for (auto const& trk : tracks) {
90  tofpid(resp_Electron.GetExpectedSignal(collision, trk),
91  resp_Muon.GetExpectedSignal(collision, trk),
92  resp_Pion.GetExpectedSignal(collision, trk),
93  resp_Kaon.GetExpectedSignal(collision, trk),
94  resp_Proton.GetExpectedSignal(collision, trk),
95  resp_Deuteron.GetExpectedSignal(collision, trk),
96  resp_Triton.GetExpectedSignal(collision, trk),
97  resp_Helium3.GetExpectedSignal(collision, trk),
98  resp_Alpha.GetExpectedSignal(collision, trk),
99  resp_Electron.GetExpectedSigma(resp, collision, trk),
100  resp_Muon.GetExpectedSigma(resp, collision, trk),
101  resp_Pion.GetExpectedSigma(resp, collision, trk),
102  resp_Kaon.GetExpectedSigma(resp, collision, trk),
103  resp_Proton.GetExpectedSigma(resp, collision, trk),
104  resp_Deuteron.GetExpectedSigma(resp, collision, trk),
105  resp_Triton.GetExpectedSigma(resp, collision, trk),
106  resp_Helium3.GetExpectedSigma(resp, collision, trk),
107  resp_Alpha.GetExpectedSigma(resp, collision, trk),
108  resp_Electron.GetSeparation(resp, collision, trk),
109  resp_Muon.GetSeparation(resp, collision, trk),
110  resp_Pion.GetSeparation(resp, collision, trk),
111  resp_Kaon.GetSeparation(resp, collision, trk),
112  resp_Proton.GetSeparation(resp, collision, trk),
113  resp_Deuteron.GetSeparation(resp, collision, trk),
114  resp_Triton.GetSeparation(resp, collision, trk),
115  resp_Helium3.GetSeparation(resp, collision, trk),
116  resp_Alpha.GetSeparation(resp, collision, trk));
117  }
118  }
119 };

133  void process(Coll const& collision, Trks const& tracks)
134  {
135  tofpidbeta.reserve(tracks.size());

137  tofpidbeta(resp_Electron.GetBeta(collision, trk),
138  resp_Electron.GetExpectedSigma(collision, trk),
139  resp_Electron.GetExpectedSignal(collision, trk),
140  resp_Electron.GetExpectedSigma(collision, trk),
141  resp_Electron.GetSeparation(collision, trk));
142  }
143  }

270  void process(aod::Collision const& collision, soa::Join<aod::Tracks, aod::TracksExtra, aod::pidRespTOF, aod::pidRespTOFbeta> const& tracks)
271  {
272 #ifdef USE_REGISTRY
273  event("vertexz")->Fill(collision.posZ());
274 #else
275  event->Fill<vertexz>(collision.posZ());
276 #endif
277 
278  for (auto i : tracks) {
279  //
280  if (i.tofSignal() < 0) { // Skipping tracks without TOF
281  continue;
282  }
283  const float tof = i.tofSignal() - collision.collisionTime();
284 #ifdef USE_REGISTRY
285  event("htofsignal")->Fill(i.p(), i.tofSignal());
286  event("htofbeta")->Fill(i.p(), i.beta());
287  //
288  expected("hexpectedEl")->Fill(i.p(), i.tofExpSignalEl());
289  expected("hexpectedEl")->Fill(i.p(), i.tofExpSignalEl());
290  expected("hexpectedMu")->Fill(i.p(), i.tofExpSignalMu());
291  expected("hexpectedPi")->Fill(i.p(), i.tofExpSignalPi());
292  expected("hexpectedKa")->Fill(i.p(), i.tofExpSignalKa());
293  expected("hexpectedPr")->Fill(i.p(), i.tofExpSignalPr());
294  expected("hexpectedDe")->Fill(i.p(), i.tofExpSignalDe());
295  expected("hexpectedTr")->Fill(i.p(), i.tofExpSignalTr());
296  expected("hexpectedHe")->Fill(i.p(), i.tofExpSignalHe());
297  expected("hexpectedAl")->Fill(i.p(), i.tofExpSignalAl());
298  //
299  timediff("htimediffEl")->Fill(i.p(), tof - i.tofExpSignalEl());
300  timediff("htimediffMu")->Fill(i.p(), tof - i.tofExpSignalMu());
301  timediff("htimediffPi")->Fill(i.p(), tof - i.tofExpSignalPi());
302  timediff("htimediffKa")->Fill(i.p(), tof - i.tofExpSignalKa());
303  timediff("htimediffPr")->Fill(i.p(), tof - i.tofExpSignalPr());
304  timediff("htimediffDe")->Fill(i.p(), tof - i.tofExpSignalDe());
305  timediff("htimediffTr")->Fill(i.p(), tof - i.tofExpSignalTr());
306  timediff("htimediffHe")->Fill(i.p(), tof - i.tofExpSignalHe());
307  timediff("htimediffAl")->Fill(i.p(), tof - i.tofExpSignalAl());
308  //
309  nsigma("hnsigmaEl")->Fill(i.p(), i.tofNSigmaEl());
310  nsigma("hnsigmaMu")->Fill(i.p(), i.tofNSigmaMu());
311  nsigma("hnsigmaPi")->Fill(i.p(), i.tofNSigmaPi());
312  nsigma("hnsigmaKa")->Fill(i.p(), i.tofNSigmaKa());
313  nsigma("hnsigmaPr")->Fill(i.p(), i.tofNSigmaPr());
314  nsigma("hnsigmaDe")->Fill(i.p(), i.tofNSigmaDe());
315  nsigma("hnsigmaTr")->Fill(i.p(), i.tofNSigmaTr());
316  nsigma("hnsigmaHe")->Fill(i.p(), i.tofNSigmaHe());
317  nsigma("hnsigmaAl")->Fill(i.p(), i.tofNSigmaAl());
318 #else
319  event->Fill<signal>(i.p(), i.tofSignal());
320  event->Fill<tofbeta>(i.p(), i.beta());
321  //
322  expected->Fill<El>(i.p(), i.tofExpSignalEl());
323  expected->Fill<Mu>(i.p(), i.tofExpSignalMu());
324  expected->Fill<Pi>(i.p(), i.tofExpSignalPi());
325  expected->Fill<Ka>(i.p(), i.tofExpSignalKa());
326  expected->Fill<Pr>(i.p(), i.tofExpSignalPr());
327  expected->Fill<De>(i.p(), i.tofExpSignalDe());
328  expected->Fill<Tr>(i.p(), i.tofExpSignalTr());
329  expected->Fill<He>(i.p(), i.tofExpSignalHe());
330  expected->Fill<Al>(i.p(), i.tofExpSignalAl());
331  //
332  timediff->Fill<El>(i.p(), tof - i.tofExpSignalEl());
333  timediff->Fill<Mu>(i.p(), tof - i.tofExpSignalMu());
334  timediff->Fill<Pi>(i.p(), tof - i.tofExpSignalPi());
335  timediff->Fill<Ka>(i.p(), tof - i.tofExpSignalKa());
336  timediff->Fill<Pr>(i.p(), tof - i.tofExpSignalPr());
337  timediff->Fill<De>(i.p(), tof - i.tofExpSignalDe());
338  timediff->Fill<Tr>(i.p(), tof - i.tofExpSignalTr());
339  timediff->Fill<He>(i.p(), tof - i.tofExpSignalHe());
340  timediff->Fill<Al>(i.p(), tof - i.tofExpSignalAl());
341  //
342  nsigma->Fill<El>(i.p(), i.tofNSigmaEl());
