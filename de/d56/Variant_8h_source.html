154 // Poor man variant class. Does not take ownership of anything passed to it.
155 // FIXME: we should really use C++17 std::variant when it
156 // comes about
157 class Variant
158 {
159  using storage_t = std::aligned_union<8, int, int64_t, const char*, float, double, bool>::type;
160 
161  public:
162  Variant(VariantType type = VariantType::Unknown) : mType{type} {}
163 
164  template <typename T>
165  Variant(T value) : mType{variant_trait_v<T>}
166  {
167  variant_helper<storage_t, decltype(value)>::set(&mStore, value);
168  }
169 
170  template <typename T>
171  Variant(std::initializer_list<T>) : mType{VariantType::Unknown}
172  {
173  static_assert(sizeof(T) == 0,
174  "brace-enclosed initializer list forbidden for Variant"
175  "\n did you accidentally put braces around the default value?");
176  }
177 
178  Variant(const Variant& other) : mType(other.mType)
179  {
180  // In case this is a string we need to duplicate it to avoid
181  // double deletion.
182  if (mType == variant_trait_v<const char*>) {
183  variant_helper<storage_t, const char*>::set(&mStore, other.get<const char*>());
184  } else {
185  mStore = other.mStore;
186  }
187  }
188 
189  Variant(Variant&& other) : mType(other.mType)
190  {
191  // In case this is a string we need to duplicate it to avoid
192  // double deletion.
193  if (mType == variant_trait_v<const char*>) {
194  mStore = other.mStore;
195  *reinterpret_cast<char**>(&(other.mStore)) = nullptr;
196  } else {
197  mStore = other.mStore;
198  }
199  }
200 
201  ~Variant()
202  {
203  // In case we allocated a string out of bound, we
204  // should delete it.
205  if (mType == variant_trait_v<const char*> || mType == variant_trait_v<char*>) {
206  free(*reinterpret_cast<void**>(&mStore));
207  }
208  }
209 
210  void operator=(const Variant& other)
211  {
212  if (mType == variant_trait_v<const char*>) {
213  variant_helper<storage_t, const char*>::set(&mStore, other.get<const char*>());
214  } else {
215  mStore = other.mStore;
216  }
217  }
218 
219  template <typename T>
220  T get() const
221  {
222  auto v = variant_trait_v<T>;
223  if (mType != variant_trait_v<T>) {
224  throw runtime_error("Mismatch between types");
225  }
226  return variant_helper<storage_t, T>::get(&mStore);
227  }
228 
229  template <typename T>
230  void set(T value)
231  {
232  return variant_helper<storage_t, T>::set(mStore, value);
233  }
234 
235  VariantType type() const { return mType; }
236 
237  private:
238  friend std::ostream& operator<<(std::ostream& oss, Variant const& val);
239  storage_t mStore;
240  VariantType mType;
241 };
