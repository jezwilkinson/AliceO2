150  template <typename T>
151  static double Eta(const array<T, 3>& mom)
152  {
153  // eta = arctanh(pz/p)
154  if (std::abs(mom[0]) < Almost0 && std::abs(mom[1]) < Almost0) { // very small px and py
155  return (double)(mom[2] > 0 ? VeryBig : -VeryBig);
156  }
157  return (double)(std::atanh(mom[2] / P(mom)));

176  template <typename T, typename U, typename V>
177  static double CPA(const array<T, 3>& posPV, const array<U, 3>& posSV, const array<V, 3>& mom)
178  {
179  // CPA = (l . p)/(|l| |p|)
180  auto lineDecay = array{posSV[0] - posPV[0], posSV[1] - posPV[1], posSV[2] - posPV[2]};
181  auto cos = dotProd(lineDecay, mom) / std::sqrt(mag2(lineDecay) * mag2(mom));
182  if (cos < -1.) {
183  return -1.;
184  }
185  if (cos > 1.) {
186  return 1.;
187  }
188  return cos;
189  }

196  template <std::size_t N, std::size_t O, std::size_t P, typename T, typename U, typename V>
197  static double CPAXY(const array<T, N>& posPV, const array<U, O>& posSV, const array<V, P>& mom)
198  {
199  // CPAXY = (r . pT)/(|r| |pT|)
200  auto lineDecay = array{posSV[0] - posPV[0], posSV[1] - posPV[1]};
201  auto momXY = array{mom[0], mom[1]};
202  auto cos = dotProd(lineDecay, momXY) / std::sqrt(mag2(lineDecay) * mag2(momXY));
203  if (cos < -1.) {
204  return -1.;
205  }
206  if (cos > 1.) {
207  return 1.;
208  }
209  return cos;
210  }

232  template <typename T, typename U, typename V>
233  static double CosThetaStar(const array<array<T, 3>, 2>& arrMom, const array<U, 2>& arrMass, V mTot, int iProng)
234  {
235  auto pVecTot = PVec(arrMom[0], arrMom[1]); // momentum of the mother particle
236  auto pTot = P(pVecTot); // magnitude of the momentum of the mother particle
237  auto eTot = E(pTot, mTot); // energy of the mother particle
238  auto gamma = eTot / mTot; // γ, Lorentz gamma factor of the mother particle
239  auto beta = pTot / eTot; // β, velocity of the mother particle
240  auto pStar = std::sqrt(sq(sq(mTot) - sq(arrMass[0]) - sq(arrMass[1])) - sq(2 * arrMass[0] * arrMass[1])) / (2 * mTot); // p*, prong momentum in the rest frame of the mother particle
241  // p* = √[(M^2 - m1^2 - m2^2)^2 - 4 m1^2 m2^2]/2M
242  // Lorentz transformation of the longitudinal momentum of the prong into the detector frame:
243  // p_L,i = γ (p*_L,i + β E*_i)
244  // p*_L,i = p_L,i/γ - β E*_i
245  // cos(θ*_i) = (p_L,i/γ - β E*_i)/p*
246  return (dotProd(arrMom[iProng], pVecTot) / (pTot * gamma) - beta * E(pStar, arrMass[iProng])) / pStar;
247  }

365  template <std::size_t N, typename T, typename U>
366  static double M2(const array<array<T, 3>, N>& arrMom, const array<U, N>& arrMass)
367  {
368  array<double, 3> momTotal{0., 0., 0.}; // candidate momentum vector
369  double energyTot{0.}; // candidate energy
370  for (auto iProng = 0; iProng < N; ++iProng) {
371  for (auto iMom = 0; iMom < 3; ++iMom) {
372  momTotal[iMom] += arrMom[iProng][iMom];
373  } // loop over momentum components
374  energyTot += E(arrMom[iProng], arrMass[iProng]);
375  } // loop over prongs
376  return energyTot * energyTot - P2(momTotal);
377  }
